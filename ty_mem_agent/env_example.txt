# TY Memory Agent 环境配置示例
# 复制此文件为 .env 并填入您的配置

# === 基础配置 ===
HOST=0.0.0.0
PORT=8080

# === LLM配置 ===
# DashScope配置（推荐）
DASHSCOPE_API_KEY=your_dashscope_api_key_here
DEFAULT_LLM_MODEL=qwen-max

# OpenAI配置（备选）
# OPENAI_API_KEY=your_openai_api_key_here
# OPENAI_BASE_URL=https://api.openai.com/v1

# === MemOS记忆系统配置 ===
MEMOS_API_BASE=https://api.openmem.net
MEMOS_API_KEY=your_memos_api_key_here

# === MCP服务API密钥 ===
# 滴滴叫车（需要申请）
DIDI_API_KEY=your_didi_api_key_here

# 高德地图 API Key（必需 - 用于所有高德地图功能）
# 申请地址: https://lbs.amap.com/api/webservice/guide/create-project/get-key
# 功能包括: 天气查询、地理编码、POI搜索、路径规划、距离测量等
AMAP_TOKEN=your_amap_api_key_here

# === 数据库配置 ===
# Redis（可选，用于会话存储）
REDIS_URL=redis://localhost:6379/0

# SQLite（默认）
DATABASE_URL=sqlite:///./ty_mem_agent.db

# === 安全配置 ===
SECRET_KEY=your_super_secret_key_change_in_production
ACCESS_TOKEN_EXPIRE_MINUTES=30

# === 日志配置（仅保留 LOG_LEVEL 作为唯一入口） ===
LOG_LEVEL=INFO
LOG_FILE=logs/ty_mem_agent.log

# === 记忆配置 ===
MEMORY_MAX_TOKENS=4000
MEMORY_RETENTION_DAYS=30
